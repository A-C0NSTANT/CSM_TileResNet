```python
class HistoryEncoder(nn.Module):
    def forward(
        self,
        history: torch.Tensor,
        padding_mask: torch.Tensor,
    ) -> torch.Tensor:
        # history: [B, L, F]
        # padding_mask: [B, L]

        token = self.embed_fields(history)

        cls = self.cls_token.expand(history.size(0), 1, -1)
        token = torch.cat([cls, token], dim=1)

        cls_mask = torch.zeros(
            history.size(0), 1,
            dtype=torch.bool,
            device=padding_mask.device,
        )
        full_mask = torch.cat([cls_mask, padding_mask], dim=1)

        token = token + self.position_embedding(
            torch.arange(token.size(1), device=token.device)
        )[None, :, :]

        output = self.encoder(
            token,
            src_key_padding_mask=full_mask,
        )

        feature = output[:, 0]
        return self.output_norm(feature)
```

```python
def forward(
    self,
    tile,
    public,
    history,
    history_padding_mask,
    legal_mask=None,
):
    tile_feature = self.tile_encoder(tile)
    public_feature = self.public_encoder(public)

    history_feature = self.history_encoder(
        history,
        history_padding_mask,
    )

    fused = torch.cat(
        [tile_feature, public_feature, history_feature],
        dim=-1,
    )

    fused = self.fusion(fused)
    logits = self.policy_head(fused)

    if legal_mask is not None:
        logits = logits.masked_fill(~legal_mask, -1e9)

    return logits
```
